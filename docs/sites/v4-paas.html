import { Quorbz } from '@quorbz/sdk'

// Initialize with your API key
const client = new Quorbz({
  apiKey: process.env.QUORBZ_API_KEY,
})

// Spawn a persistent agent
const agent = await client.agents.create({
  role:    'content-creator',
  model:   'auto',          // routes to best model
  memory: 'persistent',   // survives restarts
})

// Repurpose a YouTube video to 30+ formats
const outputs = await agent.repurpose({
  input:    'https://youtube.com/watch?v=...',
  targets: ['instagram', 'twitter', 'linkedin'],
  voice:   agent.voiceProfile,
})

// outputs.instagram — caption + hashtags
// outputs.twitter   — 8-tweet thread
// outputs.linkedin  — formatted post
console.log(outputs.instagram.caption)

API Reference Preview

Designed for how developers actually work.

Every endpoint follows REST conventions, returns clean JSON, and includes SDK helpers for Node, Python, Go, and Ruby. The docs have runnable examples, not lorem ipsum.

Authentication is a single Bearer token. Rate limits are per-tier, not per-endpoint. Errors include a human-readable message and a docs link.

POST/v1/agentsSpawn agent
POST/v1/agents/{id}/instructSend instruction
GET/v1/agents/{id}/statusGet agent state
GET/v1/agents/{id}/memoryRetrieve memory
POST/v1/pipelinesCreate pipeline
POST/v1/pipelines/{id}/runTrigger run
GET/v1/modelsList available models
DEL/v1/agents/{id}Terminate agent
